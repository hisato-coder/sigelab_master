        % 右側：アルゴリズムのポイント
        \begin{column}{0.5\textwidth}
            \small
            \textbf{アルゴリズムのポイント}
            \begin{itemize}
                \item cut-off subtractionを原始再帰的に実装する \\ 
                本問における，原始帰納的とは「値を一つ減らすpi関数という基本的な関数と
                $n$回繰り返される再帰呼び出しを用いて完全に表現され，絶対に計算が終了し，答えが得られる」
                と考えるとわかりやすい
                \item \texttt{cut\_off\_sub} 関数は，$n-1$を行いながら自身を再帰呼び出しすることで，
                $n$回繰り返される再帰呼び出しを実現．$n$回の\texttt{pi} 関数の呼び出しで$m$を$n$回減らすことができ，
                $m-n$の値を得ることができる
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}
