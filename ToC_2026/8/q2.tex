\begin{frame}
\frametitle{Q2 問題}
\begin{alertblock}{問題}

    次の λ項を β-正規形（β-normal form）まで簡約せよ。ただし，P と Q は β-正規形であるとする。

    \begin{enumerate}
        \renewcommand{\labelenumi}{(\arabic{enumi})}
        \item $\bar{\sigma}\ \bar{3}$
        \item $add\ \bar{4}\ \bar{1}$
        \item $mul\ \bar{2}\ \bar{3}$
        \item $D_1(DPQ)$
        \item $D_2(DPQ)$
        \item $DPQ\bar{0}$
    \end{enumerate}
\end{alertblock}
\end{frame}

\begin{frame}
\frametitle{Q2 定義まとめ}
\begin{alertblock}{定義}
    \begin{itemize}
    \item $\bar{\sigma} \equiv \lambda uxy.x(uxy)$
    \item $add\ \equiv \lambda mnfx.nf(mfx)$
    \item $mul\ \equiv \lambda mnfx.n(mf)x$
    \item $\bar{0} \equiv (\lambda f.(\lambda x.x))$
    \item $\bar{1} \equiv (\lambda f.(\lambda x.(fx)))$
    \item $\bar{2} \equiv (\lambda f.(\lambda x.(f(fx))))$
    \item $K \equiv \lambda xy.x$
    \item $D \equiv \lambda xyz.z(Ky)x$
    \item $D_1 \equiv \lambda x.x\bar{0}$
    \item $D_2 \equiv \lambda x.x\bar{1}$
    \end{itemize}
\end{alertblock}

\end{frame}

\begin{frame}
\frametitle{Q2 (1) 解答}
    
\begin{align*}
    \bar{\sigma}\ \bar{3} &\equiv (\lambda uxy.x(uxy))\ (\lambda f.(\lambda x.(f(f(fx))))) &(\bar{\sigma},\bar{3}の定義より)\\
    &\equiv ((\lambda u.(\lambda x.(\lambda y.(x((ux)y)))))(\lambda f.(\lambda x.(f(f(f(x))))))) \\
    &\triangleright_{1\beta} ([(\lambda f.(\lambda x.(f(f(fx)))))/u](\lambda x.(\lambda y.(x(uxy))))) \\
    &\equiv (\lambda x.(\lambda y.(x((\lambda f.(\lambda x.(f(f(fx)))))xy)))) \\
    &\triangleright_{1\beta} (\lambda x.(\lambda y.(x([x/f](f(f(fx)))y)))) \\
    &\triangleright_{1\beta} (\lambda x.(\lambda y.(x(\lambda w.[x/f][w/x](f(f(fx))))y))) \\
    &\equiv (\lambda x.(\lambda y.(x(\lambda w.[x/f](f(f(fw)))y)))) \\
    &\equiv (\lambda x.(\lambda y.(x(\lambda w.(x(x(x(w))))y)))) \\
\end{align*}
\end{frame}

\begin{frame}
\frametitle{Q2 (1) 解答続き}
\begin{align*}
    &\triangleright_{1\beta} (\lambda x.(\lambda y.(x([y/w](x(x(x(w)))))))) \\
    &\equiv (\lambda x.(\lambda y.(x(x(x(x(y))))))) \\
    &\equiv_\alpha (\lambda f.(\lambda x.(f(f(f(f(x))))))) &(yをxに,xをfに置換)\\
    &\equiv \bar{4}
\end{align*}
\end{frame}

\begin{frame}
\frametitle{Q2 (2) 解答}
    \begin{align*}
        add\ \bar{4}\ \bar{1} &\equiv (\lambda mnfx.nf(mfx))(\lambda f.(\lambda x.(f(f(f(f(x)))))))(\lambda f.(\lambda x.(f(x)))) & (add,\bar{4},\bar{1}の定義より)\\
        &\triangleright_{1\beta} ((\lambda nfx.nf(\lambda f.(\lambda x.(f(f(f(f(x))))))fx))(\lambda f.(\lambda x.(fx)))) \\
        &\triangleright_{1\beta} (\lambda fx.(\lambda f.(\lambda x.(fx)))f((\lambda f.(\lambda x.(f(f(f(fx))))))fx)) \\
        &\triangleright_{1\beta} (\lambda fx.(\lambda x.(fx))((\lambda f.(\lambda x.(f(f(f(fx))))))fx)) \\
        &\triangleright_{1\beta} (\lambda fx.f((\lambda f.(\lambda x.(f(f(f(fx))))))fx)) \\
        &\triangleright_{1\beta} (\lambda fx.f((\lambda x.(f(f(f(fx)))))x)) \\
        &\triangleright_{1\beta} (\lambda fx.f(f(f(f(fx))))) \\
        &\equiv (\lambda f.(\lambda x.(f(f(f(f(fx))))))) \\
        &\equiv \bar{5}
    \end{align*}
\end{frame}

\begin{frame}
\frametitle{Q2 (3) 解答}
    \begin{align*}
        mul\ \bar{2}\ \bar{3}
        &\equiv (\lambda mnfx.n(mf)x)\ (\lambda f.(\lambda x.(f(fx))))\ (\lambda f.(\lambda x.(f(f(fx))))) & (mul,\bar{2},\bar{3}の定義より)\\
        &\triangleright_{1\beta} (\lambda nfx.n((\lambda f.(\lambda x.(f(fx))))f)x)\ (\lambda f.(\lambda x.(f(f(fx))))) \\
        &\triangleright_{1\beta} (\lambda fx.(\lambda f.(\lambda x.(f(f(fx)))))((\lambda f.(\lambda x.(f(fx))))f)x) \\
        &\triangleright_{1\beta} (\lambda fx.(\lambda x.((\lambda f.(\lambda x.(f(fx))))f)(((\lambda f.(\lambda x.(f(fx))))f)x))x) \\
        &\triangleright_{1\beta} (\lambda fx.((\lambda f.(\lambda x.(f(fx))))f)(((\lambda f.(\lambda x.(f(fx))))f)x)) \\
        &\triangleright_{1\beta} (\lambda fx.(\lambda x.(f(fx)))(((\lambda f.(\lambda x.(f(fx))))f)x)) \\
        &\triangleright_{1\beta} (\lambda fx.f(f(((\lambda f.(\lambda x.(f(fx))))f)x))) \\
    \end{align*}
\end{frame}

\begin{frame}
\frametitle{Q2 (3) 解答続き}
    \begin{align*}
        &\triangleright_{1\beta} (\lambda fx.f(f((\lambda x.(f(fx)))((\lambda x.(f(fx)))x)))) \\
        &\triangleright_{1\beta} (\lambda fx.f(f(f(f((\lambda x.(f(fx)))x))))) \\
        &\triangleright_{1\beta} (\lambda fx.f(f(f(f(f(fx)))))) \\
        &\equiv \lambda f.\lambda x.f(f(f(f(f(fx))))) \\
        &\equiv \bar{6}
    \end{align*}
\end{frame}

\begin{frame}
    \frametitle{Q2 (4) 解答}
    \begin{align*}
        D_1(DPQ) &\equiv (\lambda x.x\ \bar{0})(DPQ) & (D_1の定義より)\\
        &\triangleright_{1\beta} [DPQ/x](x\ \bar{0}) \\
        &\equiv (DPQ)\ \bar{0} \\
        &\equiv (((\lambda xyz.z(Ky)x)P)Q)\ \bar{0} & (Dの定義より)\\
        &\triangleright_{1\beta} (([P/x](\lambda yz.z(Ky)x))Q)\ \bar{0} \\
        &\equiv ((\lambda yz.z(Ky)P)Q)\ \bar{0} \\
        &\triangleright_{1\beta} ([Q/y](\lambda z.z(Ky)P))\ \bar{0} \\
        &\equiv (\lambda z.z(KQ)P)\ \bar{0} \\
        &\equiv (\lambda z.z(KQ)P)(\lambda f.(\lambda x.x)) & (\bar{0}の定義より)\\
    \end{align*}
\end{frame}

\begin{frame}
    \frametitle{Q2 (4) 解答続き}
    \begin{align*}
        &\triangleright_{1\beta} ([(\lambda f.(\lambda x.x)/z](z(KQ)P)) \\
        &\equiv ((\lambda f.(\lambda x.x))(KQ)P) \\
        &\triangleright_{1\beta} (([KQ/f](\lambda x.x))P) \\
        &\equiv ((\lambda x.x)P) \\
        &\triangleright_{1\beta} ([P/x]x) \\
        &\equiv P
    \end{align*}
\end{frame}

\begin{frame}
    \frametitle{Q2 (5) 解答}
    \begin{align*}
        D_2(DPQ) &\equiv (\lambda x.x\ \bar{1})(DPQ) & (D_2の定義より)\\
        &\triangleright_{1\beta} [DPQ/x](x\ \bar{1}) \\
        &\equiv (DPQ)\ \bar{1} \\
        &\equiv ((\lambda xyz.z(Ky)x)PQ)\ \bar{1} & (Dの定義より)\\
        &\triangleright_{1\beta} ((\lambda yz.z(Ky)P)Q)\ \bar{1} \\
        &\triangleright_{1\beta} (\lambda z.z(KQ)P)\ \bar{1} \\
        &\triangleright_{1\beta} \bar{1}(KQ)P \\
        &\equiv (\lambda fx.(fx))(KQ)P 
    \end{align*}
\end{frame}

\begin{frame}
    \frametitle{Q2 (5) 解答続き}
    \begin{align*}
        &\triangleright_{1\beta} (\lambda x.((KQ)x))P \\
        &\triangleright_{1\beta} (KQ)P \\
        &\equiv ((\lambda xy.x)Q)P \\
        &\triangleright_{1\beta} (\lambda y.Q)P \\
        &\triangleright_{1\beta} [P/x]Q \\
        &\equiv Q
    \end{align*}
\end{frame}

\begin{frame}
    \frametitle{Q2 (6) 解答}
    \begin{align*}
        DPQ\bar{0} &\equiv ((\lambda xyz.z(Ky)x)P)Q\ \bar{0} & (Dの定義より)\\
        &\triangleright_{1\beta} (\lambda yz.z(Ky)P)Q\ \bar{0} \\
        &\triangleright_{1\beta} (\lambda z.z(KQ)P)\ \bar{0} \\
        &\triangleright_{1\beta} \bar{0}(KQ)P \\
        &\equiv (\lambda fx.x)(KQ)P & (\bar{0}の定義より)\\
        &\triangleright_{1\beta} (\lambda x.x)P \\
        &\triangleright_{1\beta} [P/x]x \\
        &\equiv P
    \end{align*}
\end{frame}
