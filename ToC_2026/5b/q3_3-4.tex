\begin{frame}[fragile]
\frametitle{Q3(3) $T$ のTuring認識不可能性 (1/4): 証明の方針}
\begin{block}{対象となる言語 $T$}
  $T = \{\langle M \rangle \mid M \text{ is a TM that accepts } w^\mathcal{R} \text{ whenever it accepts } w\}$
  （＝受理する文字列の集合 $L(M)$ が\textbf{反転について閉じている}）
\end{block}

\begin{itemize}
  \item \textbf{目標:} $T$ がTuring認識不可能であることを示す．
  \item \textbf{証明のキーポイント:} \alert{写像帰着の性質と定理5.29を活用する}
  \begin{itemize}
      \item 定理$5.29$:  $A \le_m B$ かつ$B$ がTuring認識不可能ならば $A$ もTuring認識不可能である．
      \item $\overline{A_{\mathrm{TM}}} \le_m T$ を直接示す代わりに、それと同値である \textbf{$A_{\mathrm{TM}} \le_m \overline{T}$} を示す（$A \le_m B \iff \overline{A} \le_m \overline{B}$）．
      \item すなわち、「$M$が$w$を受理する $\iff$ $L(M')$が反転について\textbf{閉じない}」となるような $M'$ を構成できれば証明完了となる．
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Q3(3) $T$ のTuring認識不可能性 (2/4): $M'$の構成}
$A_{\mathrm{TM}}$ から $\overline{T}$ への帰着関数 $f$ を計算するTM $F$ を構築する．

\begin{block}{$F$ のアルゴリズム}
  入力 $\langle M, w \rangle$ に対して：\\
  1. 以下のTM $M'$ を構築する．\\
  \quad $M'=$ 『入力 $x$ に対して：
  \begin{itemize}
      \item $x = 00$ ならば\textbf{受理}する．
      \item $x \neq 10$ ならば\textbf{受理しない}．
      \item $M$ を入力 $w$ でシミュレートし、$M$ が $w$ を受理したならば\textbf{受理}する．』
  \end{itemize}
  2. $\langle M' \rangle$ を出力する．
\end{block}
\end{frame}

\begin{frame}[fragile]
\frametitle{Q3(3) $T$ のTuring認識不可能性 (3/4): 挙動の分析}
$M$ が $w$ を受理するかどうかで $L(M')$ が変化する．\\
$T$ の条件（$w \in L(M) \implies w^\mathcal{R} \in L(M)$）について確認する．

\begin{itemize}
  \item \textbf{ケース1: $M$ が $w$ を受理する場合}
  \begin{itemize}
      \item $L(M') = \{00, 10\}$ となる．
      \item $10 \in L(M')$ だが、その反転 $10^\mathcal{R} = 01 \notin L(M')$ である．
      \item ゆえに、\alert{反転について閉じない}ため、$\langle M' \rangle \notin T$（すなわち $\langle M' \rangle \in \overline{T}$）
  \end{itemize}
  \vspace{1em}
  \item \textbf{ケース2: $M$ が $w$ を受理しない場合}
  \begin{itemize}
      \item $L(M') = \{00\}$ となる．
      \item $00^\mathcal{R} = 00 \in L(M')$ である．
      \item ゆえに、\textbf{反転について閉じている}ため、$\langle M' \rangle \in T$（すなわち $\langle M' \rangle \notin \overline{T}$）
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Q3(3) $T$ のTuring認識不可能性 (4/4): 結論}
前ページの分析により、以下の同値関係が確認できた．

\begin{block}{帰着の成立}
  したがって、\alert{$\langle M, w \rangle \in A_{\mathrm{TM}} \iff \langle M' \rangle \in \overline{T}$} が成り立つ．
\end{block}

\vspace{0.5em}
\begin{itemize}
  \item ゆえに、関数 $f$ は $A_{\mathrm{TM}}$ から $\overline{T}$ への写像帰着である（$A_{\mathrm{TM}} \le_m \overline{T}$）．
  \item 写像帰着の性質より、これは \textbf{$\overline{A_{\mathrm{TM}}} \le_m T$} と同値である．
  \item $\overline{A_{\mathrm{TM}}}$ がTuring認識不可能であるため、\alert{定理5.29より $T$ もTuring認識不可能}である．\qed
\end{itemize}
\end{frame}

\begin{frame}[allowframebreaks, fragile]
\frametitle{Q3(3): 解答例}

\textbf{証明:} $T$ がTuring認識不可能であることを示す．

$\overline{A_{\mathrm{TM}}}$ がTuring認識不可能であることから，$\overline{A_{\mathrm{TM}}} \le_m T$ を示す．
（写像帰着の性質 $A \le_m B \iff \overline{A} \le_m \overline{B}$ より，$A_{\mathrm{TM}} \le_m \overline{T}$ を示すことと同値である．）

$A_{\mathrm{TM}}$ から $\overline{T}$ への帰着関数 $f$ を計算するTuring機械 $F$ を以下のように構築する．\\
$F=$ 「入力 $\langle M, w \rangle$ に対して：
\begin{enumerate}
    \item 以下のTuring機械 $M'$ を構築する． \\
     $M'=$ 『入力 $x$ に対して：
     \begin{itemize}
         \item $x = 00$ ならば受理する．
         \item $x \neq 10$ ならば受理しない．
         \item $M$ を入力 $w$ でシミュレートし，$M$ が $w$ を受理したならば受理する．』
     \end{itemize}
    \item $\langle M' \rangle$ を出力する．」
\end{enumerate}

\framebreak % ここで自動的に次のスライドに分割されます

$M'$ の定義より，$M$ が $w$ を受理するならば $L(M') = \{00, 10\}$ となり，そうでなければ $L(M') = \{00\}$ となる．
ここで，$T$ の条件（$w \in L(M) \implies w^\mathcal{R} \in L(M)$）について確認する．

\begin{itemize}
    \item $M$ が $w$ を受理する場合：$L(M') = \{00, 10\}$ である．$10 \in L(M')$ だが，その反転である $10^\mathcal{R} = 01 \notin L(M')$ であるため，$L(M')$ は反転について閉じない．よって $\langle M' \rangle \notin T$（すなわち $\langle M' \rangle \in \overline{T}$）となる．
    \item $M$ が $w$ を受理しない場合：$L(M') = \{00\}$ である．$00^\mathcal{R} = 00 \in L(M')$ であるため，$L(M')$ は反転について閉じている．よって $\langle M' \rangle \in T$（すなわち $\langle M' \rangle \notin \overline{T}$）となる．
\end{itemize}

したがって，$\langle M, w \rangle \in A_{\mathrm{TM}} \iff \langle M' \rangle \in \overline{T}$ が成り立つ．
ゆえに，関数 $f$ は $A_{\mathrm{TM}}$ から $\overline{T}$ への写像帰着である（$A_{\mathrm{TM}} \le_m \overline{T}$）．

これは $\overline{A_{\mathrm{TM}}} \le_m T$ と同値である．$\overline{A_{\mathrm{TM}}}$ がTuring認識不可能であるため，定理5.29より $T$ もTuring認識不可能である．\qed
\end{frame}

\begin{frame}[fragile]
\frametitle{Q3(4) $OL$ のTuring認識可能性 (1/3): 方針}
\begin{block}{対象となる言語}
  $OL = \{\langle M, A \rangle \mid L(M) \cap L(A) \neq \emptyset\}$\\
  （TM $M$ と DFA $A$ が共通して受理する文字列が存在するペアの集合）
\end{block}

\begin{itemize}
  \item \textbf{目標:} $OL$ がTuring認識可能であることを示す．
  \item \textbf{証明の方針:}
  \begin{itemize}
      \item 認識可能であることが既知である \alert{$\overline{E_{\mathrm{TM}}}$} に帰着させる．
      \item ※ $E_{\mathrm{TM}}$ は空言語を受理するTMの集合なので、$\overline{E_{\mathrm{TM}}}$ は「空ではない（何か1つでも受理する）言語」を受理するTMの集合である．
      \item つまり、$OL \le_m \overline{E_{\mathrm{TM}}}$ を示せばよい．
      \item 定理5.28:$A \le_m B$ かつ$B$ がTuring認識可能ならば $A$ もTuring認識可能である．
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Q3(4) $OL$ のTuring認識可能性 (2/3): $M'$の構成}
$OL$ から $\overline{E_{\mathrm{TM}}}$ への写像帰着関数を計算するTM $F$ を以下のように構築する．

\begin{block}{$M'$ のアルゴリズム}
  $F=$ 「入力 $\langle M, A \rangle$ に対して：\\
  1. 以下のTM $M'$ を構築する．\\
  \quad $M'=$ 『入力 $x$ に対して：
  \begin{itemize}
      \item $A$ を $x$ でシミュレートし、受理しないならば\textbf{拒否}する．
      \item $M$ を $x$ でシミュレートし、受理したならば\textbf{受理}する．』
  \end{itemize}
  2. $\langle M' \rangle$ を出力する．」
\end{block}
\end{frame}

\begin{frame}[fragile]
\frametitle{Q3(4) $OL$ のTuring認識可能性 (3/3): 正当性と結論}
構成した $M'$ は、$A$ と $M$ の\textbf{両方}が受理する文字列のみを受理する．

\begin{itemize}
  \item \textbf{$M'$ の受理する言語:}
  \begin{itemize}
      \item 明らかに $L(M') = L(M) \cap L(A)$ となる．
  \end{itemize}
  \item \textbf{同値性の確認:}
  \begin{itemize}
      \item $L(M) \cap L(A) \neq \emptyset \iff L(M') \neq \emptyset$
      \item $L(M') \neq \emptyset$ ということは、$\langle M' \rangle$ は「空でない言語を受理するTM」であるため、$\langle M' \rangle \in \overline{E_{\mathrm{TM}}}$ となる．
  \end{itemize}
\end{itemize}

\vspace{0.5em}
したがって、$\langle M, A \rangle \in OL \iff \langle M' \rangle \in \overline{E_{\mathrm{TM}}}$ が成立する．\\
$OL \le_m \overline{E_{\mathrm{TM}}}$ が示されたため、$\overline{E_{\mathrm{TM}}}$がTuring認識可能であるため、定理5.28より$OL$ もTuring認識可能である．\qed
\end{frame}
